  <script>

    function updateClock() {

      const now = new Date();

      const hkNow = new Date(now.toLocaleString("en-US", { timeZone: "Asia/Hong_Kong" }));

      

      let hour = hkNow.getHours();

      let minute = hkNow.getMinutes();

      const dayIndex = hkNow.getDay();

      const days = ["SUNDAY", "MONDAY", "TUESDAY", "WEDNESDAY", "THURSDAY", "FRIDAY", "SATURDAY"];

      

      const ampm = hour >= 12 ? "PM" : "AM";

      hour = hour % 12 || 12; // convert to 12-hour format

      

      document.getElementById("hour").textContent   = hour < 10 ? "0" + hour : hour;

      document.getElementById("minute").textContent = minute < 10 ? "0" + minute : minute;

      document.getElementById("ampm").textContent   = ampm;

      document.getElementById("day").textContent    = days[dayIndex];

    }


    setInterval(updateClock, 1000);

    updateClock(); // initial call

  </script>
